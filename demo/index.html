HeltarChat web widget — demo

This page loads the bundled widget from ../dist/web.js and mounts it against a local backend. Adjust businessId + apiHost below to match your env.

The floating green bubble in the bottom-right corner is the widget. Click to open. Messages flow to POST /v1/webhooks/web/<bizId> and back through Socket.io.

Snippet customers will paste

<script src="https://<YOUR_DASHBOARD_HOST>/web-widget.js" defer></script>
<script>
  window.addEventListener('DOMContentLoaded', function () {
    HeltarChat.initBubble({
      businessId: 123,
      apiHost: 'https://app.heltar.com',
      mode: 'system', // 'light' | 'dark' | 'system' — bind to your site theme
      theme: {
        primaryColor: '#008069',
        headerTitle: 'Acme Support',
        headerSubtitle: 'We typically reply in 5 minutes',
        welcomeMessage: 'Hi! Ask us anything — a human or our AI agent will reply.',
      },
    });
  });

  // When your site's theme toggles, flip the widget live:
  HeltarChat.setMode('dark');
</script>
